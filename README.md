# tushare_cq
tushare command call

tushare挖地兔 命令调用快捷方式
#-----------------------------------------
python tuShare.py 调用方法名 [参数1] [参数2]
#-----------------------------------------
如：获取实时分笔方法
# 代码
import tushare as ts
df = ts.get_realtime_quotes('159915')

# 命令 
python tuShare.py get_realtime_quotes symbols=159915
